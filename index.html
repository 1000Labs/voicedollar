export default async function handler(req, res) { if (req.method !== 'POST') { return res.status(405).json({ error: 'Method not allowed' }); } const { inputs } = req.body; if (!inputs || inputs.length === 0) { return res.status(400).json({ error: 'Text required' }); } try { // FREE gTTS API - Works everywhere, no keys needed const gttsUrl = `https://translate.google.com/translate_tts?ie=UTF-8&tl=hi&client=tw-ob&q=${encodeURIComponent(inputs.slice(0,100))}`; const response = await fetch(gttsUrl); if (!response.ok) { throw new Error('gTTS failed'); } const arrayBuffer = await response.arrayBuffer(); res.setHeader('Content-Type', 'audio/mp3'); res.setHeader('Cache-Control', 'public, max-age=3600'); res.send(Buffer.from(arrayBuffer)); } catch (error) { // Fallback to English if Hindi fails const fallbackUrl = `https://translate.google.com/translate_tts?ie=UTF-8&tl=en&client=tw-ob&q=${encodeURIComponent(inputs.slice(0,100))}`; const fallbackResponse = await fetch(fallbackUrl); const fallbackBuffer = await fallbackResponse.arrayBuffer(); res.setHeader('Content-Type', 'audio/mp3'); res.send(Buffer.from(fallbackBuffer)); } }
